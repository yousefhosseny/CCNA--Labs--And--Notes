
## مثال تكوين نموذجي يجمع كل الخطوات (مبسّط)

```shell
conf t
! ----- إعداد منافذ المستخدمين -----
vlan 10
 name USERS
vlan 999
 name UNUSED

interface range GigabitEthernet0/1 - 24
  description USER_ACCESS
  switchport mode access
  switchport access vlan 10
  spanning-tree portfast
  spanning-tree bpduguard enable
  switchport port-security
  switchport port-security maximum 2
  switchport port-security violation restrict
  switchport port-security mac-address sticky
exit

! ----- منافذ فاضية -----
interface range GigabitEthernet0/25 - 28
  description UNUSED - SHUTDOWN
  switchport mode access
  switchport access vlan 999
  shutdown
exit

! ----- Uplink / Trunk -----
interface GigabitEthernet0/48
  description UPLINK_TO_CORE
  switchport trunk encapsulation dot1q    ! if supported
  switchport mode trunk
  switchport nonegotiate
  switchport trunk allowed vlan 10,20,30
  switchport trunk native vlan 99
exit

! ----- Tag native VLAN globally (إن أمكن) -----
vlan dot1q tag native
end
write memory
```
